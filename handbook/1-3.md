```haskell
item :: Parser Char
item = do
    cs <- get
    case cs of
        ""      -> mzero
        (c:cs') -> put cs' >> return c

satisfy :: (Char -> Bool) -> Parser Char
satisfy f = do
    c <- item
    if f c
        then return c
        else mzero

char :: Char -> Parser Char
char c = satisfy (c ==)

spaces :: Parser String
spaces = many $ char ' '

string :: String -> Parser String
string ""     = return ""
string (c:cs) = do
    char c
    string cs
    return (c:cs)

oneOf :: String -> Parser Char
oneOf cs = satisfy (\c -> elem c cs)
```
